def edit_owner():    """ Edits the owner's information. """    username = get_jwt_identity()    owner = Owner.query.filter_by(username=username).first()    if owner:        password = request.form.get('password')        if password:            salt, hashed_password = secure_password(password)            owner.password = hashed_password            owner.salt = salt        owner.email = request.form.get('email')        owner.phone_number = request.form.get('phone_number')        db.session.commit()        return jsonify("Success")    else:        return jsonify("Owner Does Not Exist")
