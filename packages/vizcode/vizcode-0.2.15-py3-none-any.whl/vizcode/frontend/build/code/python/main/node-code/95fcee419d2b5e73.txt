def get_charities_for_fund():    fund_name = request.form.get('fund_name')    sql_query = '''select organization_name, organization_website                    from organizations, funding                    where organizations.email=funding.organization_id                    and funding.fund_id='{}';'''.format(fund_name)    conn = db.engine.connect().connection    df = pd.read_sql(sql_query, conn)    return jsonify(charities=list(df.T.to_dict().values()))
