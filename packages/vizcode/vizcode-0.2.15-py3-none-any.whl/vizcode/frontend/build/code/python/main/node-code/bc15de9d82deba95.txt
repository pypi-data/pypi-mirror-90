def emory_scraper(url):    response = requests.get(url, verify=False)    html = response.text    soup = BeautifulSoup(html, "lxml")    article_title = soup.find('h1', {'class': 'entry-title'}).getText()    elements = soup.findAll('span', attrs={'style': 'font-weight: 400;'})    article_text = ''.join(e.getText() for e in elements)    info = {"title": article_title.lower(), "content": article_text.lower()}    return info
