def get_article_text(link):    response = requests.get(link, verify=False)    soup = BeautifulSoup(response.text, "lxml")    if 'cnn' in link:        elements = soup.find_all("div", {"class": 'zn-body__paragraph'})    # other newspapers    else:        elements = soup.findAll('p')    texts = [element.text.strip() for element in elements]    article = " ".join(texts)    return article
